 29   /**

 31    * @function
 32    * @param {object} db (The sqlite3 db object from module node-sqlite3)
 33    * @param {array} keys (The object keys to be save in the info table)
 34    * @returns {object} (The promise with the total number of keys saved or error)
 35    * @memberOf info
 36    */
 37   info.setInfoKeys = function(db, keys) {
 38     const replaceQuery = "REPLACE INTO info (key,value) VALUES";
 39     let valueQuery = "";
 40 
 41     _.forEach(keys, (valueInfo, idx) => {
 42       const keyValuePair = _.map(valueInfo, (value, key) => {
 43         // Replace single double quotes with two double quotes
 44         const sqlValidKey = `"${key.replace(/"/g, '""')}"`;
 45         const sqlValidValue = `"${JSON.stringify(value).replace(/"/g, '""')}"`;
 46 
 47         return `(${sqlValidKey},${sqlValidValue})`;
 48       });
 49 
 50       if (keyValuePair.length) {
 51         const commaTuple = (idx < keys.length - 1) ? "," : ";";
 52         valueQuery += keyValuePair[0] + commaTuple;
 53       }
 54     });
 55 
 56     if (valueQuery !== "")
 57       return db.runAsync(replaceQuery + valueQuery, [])
 58               .then(() => {
 59                 return Promise.resolve(keys.length);
 60               });
 61     return Promise.resolve(0);
 62   };

 75   /**
 76    * Checks if info table exists.
 77    * @function
 78    * @param {object} db (The sqlite3 db object from module node-sqlite3)
 79    * @returns {object} (The promise with true or false)
 80    */
 81   function checkInfoTable(db) {
 82     return db.allAsync("SELECT name FROM sqlite_master WHERE type='table'", [])
 83       .then((rows) => {
 84         const search = _.find(rows, {"name": sqliteConstants.DATABASE_INFO_TABLE_NAME});
 85         if (search === undefined)
 86           return Promise.resolve(false);
 87         else return Promise.resolve(true);
 88       });
 89   }
